historicinvasions_debug_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/historicinvasions_pink.dds"
	}
	desc = historicinvasions_debug_decision_desc
	selection_tooltip = historicinvasions_debug_tooltip
	confirm_text = historicinvasions_confirm
	decision_group_type = historicinvasions_debug_decisions
	sort_order = 10

	is_shown = {
		debug_only = yes
		is_ai = no

		NOT = { # only once
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:historicinvasions_debug_decision
			}
		}
	}

	is_valid = {
		OR = {
			always = yes

			capital_county.empire = primary_title

			primary_title = {
				is_landless_type_title = yes
			}
			
			AND = {
				completely_controls_region = special_yelu_conquest_region_total
				completely_controls_region = special_jochid_conquest_region_white_horde
			}
		}
	}

	is_valid_showing_failures_only = {
		#is_available_adult_or_is_commanding = yes
	}

	effect = {
		add_to_global_variable_list = { # you may only enact the decision once
			name = unavailable_unique_decisions
			target = flag:historicinvasions_debug_decision
		}
	}
	ai_check_interval = 0
	ai_potential = { always = no }
}

historicinvasions_enable_debug_decisions = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/historicinvasions_green.dds"
	}
	desc = historicinvasions_enable_debug_decisions_desc
	selection_tooltip = historicinvasions_debug_tooltip
	confirm_text = historicinvasions_confirm
	decision_group_type = historicinvasions_debug_decisions
	should_create_alert = { always = no }
	sort_order = 5

	is_shown = {
		debug_only = yes
		is_ai = no

		is_ruler = yes
		is_landed = yes
		exists = dynasty

		OR = {
			NOT = { exists = global_var:historicinvasions_debug_decisions }
			global_var:historicinvasions_debug_decisions = 0
		}
	}

	is_valid = {
		always = yes
		trigger_if = {
			limit = { NOT = { exists = title:e_timurid } } # this title ought to be present in every save game started with this mod
			custom_tooltip = {
				text = historicinvasions_not_save_game_compatible
				#always = no
			}
		}
	}

	effect = {
		if = {
			limit = { exists = global_var:historicinvasions_debug_decisions }
			change_global_variable = {
				name = historicinvasions_debug_decisions
				add = 1
			}
		}
		else = {
			set_global_variable = {
				name = historicinvasions_debug_decisions
				value = 1
			}
		}
	}
	ai_check_interval = 0
	ai_potential = { always = no }
}
