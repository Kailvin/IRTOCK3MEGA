city_state_government = {
	primary_holding = city_holding
	government_rules = {
		rulers_should_have_dynasty = yes
		royal_court = yes
		legitimacy = yes
	}
	valid_holdings = { castle_holding tribal_holding city_holding trade_center_holding }
	required_county_holdings = { city_holding castle_holding }

	ai = {
		use_legends = yes
	}
	flag = government_is_feudal
	flag = government_is_city_state
	flag = government_is_settled
	vassal_contract_group = feudal_vassal

	character_modifier = {
		domain_limit = -200
		domain_tax_mult = 1
		building_slot_add = 3
	}
	color = { 0 30 60 }

	opinion_of_liege = {
		if = {
			limit = {
				scope:liege = {
					has_variable = ecostability
				}
			}
			scope:vassal = {
				if = {
					limit = {
						scope:liege = {
							government_has_flag = government_is_city_state
						}
					}
	
					if = {
						limit = {
							scope:liege = {
								var:ecostability > 50
							}
						}
							if = {
								limit = {
									scope:liege = {
										var:ecostability > 100
									}
								}
								value = 20
							}
							else_if = {
								limit = {
									scope:liege = {
										var:ecostability < 100
									}
								}
								value = 10
							}
					}
					else_if = {
						limit = {
							scope:liege = {
								var:ecostability < -50
							}
						}
							if = {
								limit = {
									scope:liege = {
										var:ecostability > -100
									}
								}
								value = -10
							}
							else_if = {
								limit = {
									scope:liege = {
										var:ecostability < -100
									}
								}
								value = -20
							}
					}
					
				}
			}
		}
	}

	opinion_of_liege_desc = {
		first_valid = {
			desc = "ecostability_name"
		}
	}
		
}

	}
	flag = government_is_feudal
	flag = government_is_gupta
	flag = government_is_settled
	vassal_contract_group = gupta_vassal
	color = { 300 155 0 }

	opinion_of_liege = {
		scope:vassal = {
			if = {
				limit = {
					OR = {
						has_trait = brahmin
						has_trait = kshatriya
						has_trait = vaishya
						has_trait = shudra
					}
				}
				if = {
					limit = {
						scope:liege = {
							OR = {
								has_trait = brahmin
								has_trait = kshatriya
								has_trait = vaishya
								has_trait = shudra
							}
						}
					}
					if = {
						limit = {
							liege = {
								caste_purity > 0.85
							}
						}
						value = 30
					}
					else_if = {
						limit = {
							liege = {
								caste_purity > 0.7
							}
						}
						value = 10
					}
					else_if = {
						limit = {
							liege = {
								caste_purity >= 0.5
							}
						}
						value = 5
					}
					else_if = {
						limit = {
							liege = {
								caste_purity > 0.4
							}
						}
						value = -5
					}
					else_if = {
						limit = {
							liege = {
								caste_purity > 0.3
							}
						}
						value = -10
					}
					else = {
						value = -20
					}
				}
				else = {
					value = -30
				}
			}
		}
	}

	opinion_of_liege_desc = {
		first_valid = {
			desc = caste_opinion
		}
	}
}
