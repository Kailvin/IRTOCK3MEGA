# A title is inherited by a character
# root = the new holder
# scope:title = the title that changes hands
# scope:previous_holder = previous holder. Should be dead
on_title_gain_inheritance = {
	on_actions = {
		invader_on_title_gain_inheritance
	}
}

# delayed because landless characters can't have dread
invader_on_title_gain_inheritance = {
	trigger = {
		has_character_flag = gain_invader_succession_dread
		has_character_modifier = historicinvasions_plot_armor
	}
	effect = {
		add_dread = high_dread # 80
	}
}
