on_game_start_after_lobby = {
	on_actions = { irtock3_on_game_start_after_lobby }
}
irtock3_on_game_start_after_lobby = {
	effect = {
	# IRToCK3: disable Nicene after the Council of Chalcedon.
	if = {
		limit = {
			game_start_date >= 451.8.25
		}
		faith:armenian_apostolic = {
			remove_doctrine = unavailable_doctrine
		}
		faith:nestorian = {
			remove_doctrine = unavailable_doctrine
		}
		faith:coptic = {
			remove_doctrine = unavailable_doctrine
		}
		faith:syriac = {
			remove_doctrine = unavailable_doctrine
		}
		faith:chalcedonian = {
			remove_doctrine = unavailable_doctrine
		}
		faith:nicene = {
			add_doctrine = unavailable_doctrine
		}
	}
	# IRToCK3: disable the Seven Houses mechanic for Persia.
	set_global_variable = {
		name = sevenhouses_dead
		value = yes
	}
	every_ruler = {
		limit = {
			capital_county.title_province = { geographical_region = world_europe_west }
			culture = culture:greek
		}
		set_culture = culture:galatian
		every_courtier_or_guest = {
            limit = { culture = culture:greek }
            set_culture = culture:galatian
        }
	}
	every_county = {
		limit = {
			title_province = { geographical_region = world_india }
			faith = faith:hellenic_pagan
		}
		set_county_faith = faith:indo_hellenism

	}
	every_ruler = {
		limit = {
			OR = {
				capital_county.title_province = { geographical_region = world_middle_east }
				capital_county.title_province = { geographical_region = world_middle_east_persia }
			}
			faith = faith:hellenic_pagan
		}
		set_character_faith = faith:mithraic_mysteries
		every_courtier_or_guest = {
            limit = { faith = faith:hellenic_pagan }
            set_character_faith = faith:mithraic_mysteries
        }
	}
	every_county = {
		limit = {
			OR = {
				title_province = { geographical_region = world_middle_east }
				title_province = { geographical_region = world_middle_east_persia }
			}
			faith = faith:hellenic_pagan
		}
		set_county_faith = faith:mithraic_mysteries

	}
	every_county = {
		limit = {
			OR = {
				title_province = { geographical_region = world_africa_north_east }
			}
			faith = faith:egyptian_pagan
		}
		add_faith_minority_effect = {
			FAITH = faith:nicene
			SIZE = large
			COUNTY = THIS
		}

	}
	}
}
