on_game_start = {
	on_actions = {
		on_roman_government_start
	}
	on_actions = {
		delay = { days = 1 }
		on_government_change_start
	}
}

on_government_change_start = {
	effect = {
		if = {
			limit = {
				title:e_byzantium = {
					is_title_created = yes
				}
			}
			title:e_byzantium.holder ?= {
				every_vassal_or_below = {
					limit = {
						OR = {
						has_government = feudal_government
						has_government = republic_government
						}
					}
					if = {
						limit = {
							highest_held_title_tier > tier_county
							has_dlc_feature = roads_to_power
						}
						change_government = administrative_government
						if = {
							limit = { 
								NOT = { any_held_title = { is_noble_family_title = yes } } 
								highest_held_title_tier >= tier_duchy
							}
							if = {
								limit = { is_house_head = no }
								found_cadet_house_decision_effect = { CHARACTER = root }
							}
							create_noble_family_effect = yes
							domicile ?= { set_up_domicile_estate_effect = yes }
						}
					}
					else_if = {
						limit = {
							highest_held_title_tier > tier_county
						}
						change_government = imperial_government
					}
					else_if = {
						limit = {
							highest_held_title_tier = tier_county
						}
						change_government = roman_government
					}
				}
			}
		}
		
		
		if = {
			limit = {
				title:e_persia = {
					is_title_created = yes
				}
			}
			title:e_persia.holder ?= {
				every_vassal_or_below = {
					limit = {
						has_government = feudal_government
						
					}
					change_government = eranshar_government
				}
			}
		}
		if = {
			limit = {
				title:e_western_roman_empire = {
					is_title_created = yes
				}
			}
			title:e_western_roman_empire.holder ?= {
				every_vassal_or_below = {
					limit = {
						OR = {
						has_government = feudal_government
						has_government = republic_government
						}

							}
							if = {
								limit = { is_house_head = no }

							}
							create_noble_family_effect = yes
							domicile ?= { set_up_domicile_estate_effect = yes }
						}
					}
					else_if = {
						limit = {
							highest_held_title_tier > tier_county
						}
						change_government = imperial_government
					}
					else_if = {
						limit = {
							highest_held_title_tier = tier_county
						}
						change_government = roman_government
					}
				}
			}
		}
		if = {
			limit = {
				title:e_roman_empire = {
					is_title_created = yes
				}
			}
			title:e_roman_empire.holder ?= {
				every_vassal_or_below = {
					limit = {
						OR = {
						has_government = feudal_government
