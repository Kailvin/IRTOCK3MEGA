form_first_bulgarian_empire_decision_effects = {
	add_to_global_variable_list = {
		name = unavailable_unique_decisions
		target = flag:form_first_bulgarian_empire_decision
	}
	
	add_prestige = massive_prestige_gain

	title:k_bulgaria.holder = { save_scope_as = first_bulgarian_emperor }

	create_title_and_vassal_change = {
		type = created
		save_scope_as = title_change
		add_claim_on_loss = no
	}
	title:e_bulgaria = {
		change_title_holder = {
			holder = scope:first_bulgarian_emperor
			change = scope:title_change
		}
	}
	resolve_title_and_vassal_change = scope:title_change

	title:e_bulgaria.holder = {
		every_held_title = {
			limit = {
				tier = tier_kingdom
			}
			title:e_bulgaria.holder = {
				destroy_title = prev
			}
		}
	}

	title:k_bulgaria = { add_to_temporary_list = hi_e_bulgaria_list }
	title:k_dacia = { add_to_temporary_list = hi_e_bulgaria_list }
	title:k_moldavia = { add_to_temporary_list = hi_e_bulgaria_list }
	every_in_list = {
		list = hi_e_bulgaria_list
		every_in_de_jure_hierarchy = {
			limit = { tier = tier_kingdom }
			set_de_jure_liege_title = title:e_bulgaria
		}
	}
	
	title:k_croatia = { add_to_temporary_list = hi_e_carpathia_list }
	title:k_bosnia = { add_to_temporary_list = hi_e_carpathia_list }
	title:k_serbia = { add_to_temporary_list = hi_e_carpathia_list }
	every_in_list = {
		list = hi_e_carpathia_list
		every_in_de_jure_hierarchy = {
			limit = { # only kingdoms that are de jure part of e_byzantium
				target_is_de_jure_liege_or_above = title:e_byzantium
				tier = tier_kingdom
			}
			set_de_jure_liege_title = title:k_hungary.de_jure_liege
		}
	}

	# legend
	gain_heroic_legend_seed_tooltip_effect = yes
	hidden_effect = { legend_seed_new_title_effect = yes }
}
