# spawn invader, before story
spawn_gediminid_invader_effect = {
	random_county_in_region = {
		region = special_$INVADER_DYNN$_conquest_region_start
		title_province = { save_scope_as = invader_spawn_location }
	}

	# INVADER
	create_character = {
		template = gediminas_gediminid_character_template
		location = scope:invader_spawn_location
		save_scope_as = invader
	}

	scope:invader = {
		add_gold = 5000
		add_prestige = 1250
		add_piety = 150
		
		dynasty = {
			set_dynasty_name = hi_dynn_gediminid
			set_coa = title:d_lithuanians

			add_dynasty_prestige_level = 4
			add_dynasty_prestige = 3000
			
			$INVADER_DYNN$_add_dynasty_perks_effect = yes
		}
		house = {
			set_house_name = hi_dynn_gediminid
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}

		limit = { has_character_flag = is_first_invader_flag }

		spawn_gediminid_army_effect = yes
	}
	spawn_gediminid_army_effect = yes

	if = {
		limit = { game_start_date < 1000.1.1 }
		spawn_gediminid_army_effect = yes
	}
}

gediminid_establish_plc_decision_effect = {
	# PLC de jure changes
	every_kingdom = {
		limit = {
			any_de_jure_county = {
				title_province = { geographical_region = special_gediminid_plc_region }
				percent >= 0.6
			}
		}
		add_to_temporary_list = gediminid_establish_plc_list
	}
	every_in_list = {
		list = gediminid_establish_plc_list
		every_in_de_jure_hierarchy = {
			limit = { tier = tier_kingdom }
			set_de_jure_liege_title = title:e_gediminid
		}
	}

	# other effects
	primary_title = { set_title_name = hi_commonwealth } # changes name & adjective
	dynasty = { add_dynasty_prestige = 1500 }

	# de jure cleanup of Pomerania
	if = {
		limit = {
			title:k_pomerania.de_jure_liege = title:e_wendish_empire
			#title:k_pomerania.de_jure_liege = { any_in_de_jure_hierarchy = { tier = tier_kingdom count <= 2 } }
		}
		title:k_pomerania = { set_de_jure_liege_title = title:d_east_franconia.de_jure_liege.de_jure_liege }
	}

	### only once
	hidden_effect = {
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:gediminid_establish_plc_decision
		}
	}
}
