# spawn invader, before story
spawn_jabalid_invader_effect = {
	random_county_in_region = {
		region = special_$INVADER_DYNN$_conquest_region_start
		title_province = { save_scope_as = invader_spawn_location }
	}

	# family
	create_character = {
		template = muhammad_jabalid_character_template
		dynasty = generate
		location = scope:invader_spawn_location
		save_scope_as = muhammad_jabalid
	}
	create_character = {
		template = sulayman_jabalid_character_template
		dynasty = inherit
		father = scope:muhammad_jabalid
		location = scope:invader_spawn_location
		save_scope_as = sulayman_jabalid
	}

	# INVADER
	create_character = {
		template = sinan_jabalid_character_template
		dynasty = inherit
		father = scope:sulayman_jabalid
		location = scope:invader_spawn_location
		save_scope_as = invader
	}

	scope:invader = {
		add_gold = 500
		add_prestige = 500
		add_piety = 200
		
		dynasty = {
			set_dynasty_name = hi_dynn_jabalid
			set_coa = title:k_jabalid

			add_dynasty_prestige_level = 1
			add_dynasty_prestige = 1500
			
			$INVADER_DYNN$_add_dynasty_perks_effect = yes
		}
		house = {
			set_house_name = hi_dynn_jabalid
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}

jabalid_add_dynasty_perks_effect = {
	add_dynasty_perk = guile_legacy_1
}

# Make sure invader has troops
spawn_jabalid_troops_effect = {
	if = {
		limit = { has_character_flag = is_first_invader_flag }
		spawn_jabalid_army_effect = yes
	}
	spawn_jabalid_army_effect = yes
}

			type = bowmen

		}
