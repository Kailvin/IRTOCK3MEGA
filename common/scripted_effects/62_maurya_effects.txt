# spawn invader, before story
spawn_maurya_invader_effect = {
	random_county_in_region = {
		region = special_$INVADER_DYNN$_conquest_region_start
		title_province = { save_scope_as = invader_spawn_location }
	}

	# INVADER
	create_character = {
		template = chandragupta_maurya_character_template
		dynasty = generate
		location = scope:invader_spawn_location
		save_scope_as = invader
	}

	scope:invader = {
		add_gold = 5000
		add_prestige = 7500
		add_piety = 400
		
		dynasty = {
			set_dynasty_name = hi_dynn_maurya
			set_coa = title:e_maurya

			add_dynasty_prestige_level = 3
			add_dynasty_prestige = 5000
			
			$INVADER_DYNN$_add_dynasty_perks_effect = yes
		}
		house = {
			set_house_name = hi_dynn_maurya
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}

maurya_add_dynasty_perks_effect = {
	add_dynasty_perk = erudition_legacy_1
	add_dynasty_perk = law_legacy_1
	if = {
		limit = { has_ep1_dlc_trigger = yes }
		add_dynasty_perk = ep1_culture_legacy_1
	}
	else = {}
}
