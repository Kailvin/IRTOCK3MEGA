# triggered via story
komnenos_maintenance_effect = {
	spawn_komnenos_invader_effect = { # only applies bonuses, always uses existing character
		INVADER_DYNN = $INVADER_DYNN$
	}
	apply_invader_modifiers_effect = {
		INVADER_DYNN = $INVADER_DYNN$
	}
	spawn_invader_servants_effect = {
		INVADER_CHAR = $INVADER_CHAR$
	}
	trigger_event = {
		id = $INVADER_DYNN$_invasion.0100
		days = 1
	}
}

spawn_komnenos_invader_effect = {
	scope:story.story_owner = {
		save_scope_as = invader
		add_prestige = 500
		add_piety = 400

		add_perk = stalwart_leader_perk
		add_perk = hard_rule_perk
		
		dynasty = {
			add_dynasty_prestige_level = 1
			add_dynasty_prestige = 1250
			
			$INVADER_DYNN$_add_dynasty_perks_effect = yes
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}
