RICE_harranian_pagans_still_around_trigger = {
	OR = {
		title:c_harran = {
			faith = faith:harranian_pagan
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:RICE_harran_moon_temple_destroyed
			}
		}		
	}
}
