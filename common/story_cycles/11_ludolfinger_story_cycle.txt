# Story for the ludolfinger Invasion
# originally made by Petter Vilberg
# adjusted by me

story_ludolfinger_invasion = {
	on_setup = {
		historicinvasions_story_setup_effect = { # global start variable, story variables, conditional dissolution protection
			INVADER_DYNN = ludolfinger # for story start global variable
			INVADER_TITLE = title:k_east_francia # title the invader spawns with
		}
		story_owner = {
			if = { # kingdom of East Francia is not created
				limit = {
					NOT = { exists = title:k_east_francia.holder }
				}
				ludolfinger_maintenance_effect = {
					INVADER_DYNN = ludolfinger
					INVADER_CHAR = scope:story.story_owner
					INVADER_TITLE = scope:story.var:invader_title_var
					LANDLESS_YES_NO = no
					STARTING_TIER = duchy
				}
			}
			else = { # east francia still exists
				# shorter version of the family effect
				apply_invader_modifiers_effect = {
					INVADER_DYNN = ludolfinger
				}
				add_perk = stalwart_leader_perk
				add_perk = flexible_truces_perk

				spawn_invader_servants_effect = {
					INVADER_CHAR = scope:story.story_owner
				}
				form_the_ludolfinger_title_effect = {
					INVADER_DYNN = ludolfinger
					INVADER_CHAR = scope:story.story_owner
					INVADER_TITLE = title:k_east_francia
					LANDLESS_YES_NO = no
				}
			}
			trigger_event = {
				id = ludolfinger_invasion.0200
				months = 1
			}
		}
	}

	on_end = {
		invader_story_on_end_effect = { INVADER_DYNN = ludolfinger }
	}

	on_owner_death = {
		# random chance to set variable to end story after ruler N dies, depending on how many story owners there have been in total
		random = {
			chance = 0
			modifier = {
				add = 0
				var:story_owner_counter >= 1 # end story after death of ruler N?
			}
			modifier = {
				add = 100
				var:story_owner_counter >= 2 # end story after death of ruler N?
			}
			set_variable = {
				name = end_of_invasion_story
				value = yes
			}
		}
		invader_story_on_death_effect = { INVADER_DYNN = ludolfinger }
	}

	# maintenance
	effect_group = {
		days = historicinvasions_story_maintenance_value

		trigger = { always = yes }

		# make sure they have money on hand
		triggered_effect = {
			trigger = { story_owner = { gold < 0 } }
			effect = { story_owner = { add_gold = 500 } }
		}

		# end the story prematurely, because the invader has lost their #th offensive war
		triggered_effect = {
			trigger = {
				NOT = { has_game_rule = historicinvasions_invasion_ends_after_defeat_disabled }
			}
			effect = {
				historicinvasions_invasion_ends_after_defeat_effect = yes
			}
		}
	}

	#Start a new war if none is ongoing
	effect_group = {
		days = { 120 240 }

		trigger = { story_owner = { is_ai = yes } }

		first_valid = {
			triggered_effect = {
				trigger = {
					story_owner = { has_character_flag = invasion_ends_after_player_win }
				}
				effect = {
					debug_log = "ending ludolfinger story because a player was able to defend against the invader"
					end_story = yes
				}
			}
			triggered_effect = {
				trigger = { story_owner = { should_invader_declare_war_trigger = { INVADER_TITLE = scope:story.var:invader_title_var } } }
				effect = {
					story_owner = {
						war_target_evaluation_and_declaration_effect = {
							INVADER_DYNN = ludolfinger
							INVADER_CHAR = scope:story.story_owner
							INVADER_REGION = core
							INVADER_TARGET_TITLE_TIER = kingdom
							INVADER_RANGE = neighboring
							INVADER_LIEGE_RESTRICTIONS = "NOT = { this = title:k_papal_state.holder }"
						}
					}
				}
			}
		}
	}

	# Story handling in case of succession weirdness
	effect_group = {
		days = historicinvasions_succession_weirdness_value

		trigger = { NOT = { story_owner = { has_title = scope:story.var:invader_title_var } } }

		# either transfer story ownership or end story if the title is not created
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				hi_check_for_valid_story_owner_effect = {
					INVADER_TITLE = scope:story.var:invader_title_var
					STORY_COUNTER_INCREASE = 0
				}
			}
		}

		triggered_effect = {
			trigger = { exists = title:e_hre.holder }
			effect = {
				debug_log = "ending ludolfinger story due to succession weirdness trigger 3"
				end_story = yes
			}
		}
	}

	# restore the HRE through story
	effect_group = {
		days = 185

		trigger = {
			exists = title:e_hre
			story_owner = {
				#todo none held titles are being contested instead
				NOT = { any_held_title = { is_contested = yes } }
				#completely_controls_region = special_ludolfinger_conquest_region_core
				OR = {
					AND = {
						completely_controls = title:k_east_francia
						completely_controls = title:k_bavaria
						completely_controls = title:k_italy
					}
					any_county_in_region = {
						region = special_ludolfinger_conquest_region_core
						percent > 0.90
						holder.top_liege = scope:story.story_owner
					}
				}
			}
		}

		triggered_effect = {
			trigger = {
				NOT = { exists = title:e_hre.holder } # e_hre is handed out afterwards
			}

			effect = {
				debug_log = "ending ludolfinger story because the conquest goal has been completed and the HRE formed"
				story_owner = { hi_restore_hre_decision_effect = yes }
				end_story = yes
			}
		}
	}
}
