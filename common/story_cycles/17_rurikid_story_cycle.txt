# Story for Rurikid dynasty
# originally made by Petter Vilberg
# adjusted by me

story_rurikid_invasion = {
	on_setup = {
		historicinvasions_story_setup_effect = { # global start variable, story variables, conditional dissolution protection
			INVADER_DYNN = rurikid # for story start global variable
			INVADER_TITLE = title:e_rurikid # title the invader spawns with
		}
		story_owner = {
			rurikid_maintenance_effect = {
				INVADER_DYNN = rurikid
				INVADER_CHAR = scope:story.story_owner
				INVADER_TITLE = scope:story.var:invader_title_var
				LANDLESS_YES_NO = no
				STARTING_TIER = duchy
			}
			trigger_event = {
				id = rurikid_invasion.0200
				months = 1
			}
		}
		# culture stuff
		culture:norse = {
			change_cultural_acceptance = {
				target = culture:russian
				value = miniscule_positive_culture_acceptance # miniscule = 2%, low = 5%, medium = 10%
				desc = cultural_acceptance_gain_event
			}
		}
	}

	on_end = {
		invader_story_on_end_effect = { INVADER_DYNN = rurikid }

		### realm conversion event chain
		scope:story.story_owner = {
			if = {
				limit = { is_alive = yes }
				create_story = story_rurikid_conversion
			}
			else_if = {
				limit = { exists = scope:story.story_owner.player_heir }
				scope:story.story_owner.player_heir = { create_story = story_rurikid_conversion }
			}
		}

		### attempt to switch over to local title
		if = {
			limit = {
				story_owner = {
					is_alive = no
					exists = player_heir
				}
			}
			story_owner.player_heir = { save_scope_as = replacement_title_holder }
		}
		if = {
			limit = {
				hi_can_story_switch_to_local_title_trigger = {
					INVADER_TITLE = e_rurikid # to be replaced by
					LOCAL_TITLE = e_russia
				}
			}
			hi_story_switches_to_local_title_effect = {
				INVADER_TITLE = e_rurikid # to be replaced by
				LOCAL_TITLE = e_russia
			}
			title:e_russia = {
				set_title_name = e_rurikid
				set_coa = e_rurikid
			}
		}

		destroy_titular_invader_title_effect = yes
	}

	on_owner_death = {
		# random chance to set variable to end story after ruler N dies, depending on how many story owners there have been in total
		random = {
			chance = 0
			modifier = {
				add = 0
				var:story_owner_counter >= 1 # end story after death of ruler N?
			}
			modifier = {
				add = 10
				var:story_owner_counter >= 2
			}
			modifier = {
				add = 10
				var:story_owner_counter >= 3
			}
			modifier = {
				add = 60
				var:story_owner_counter >= 4
			}
			modifier = {
				add = 20
				var:story_owner_counter >= 5
			}
			set_variable = {
				name = end_of_invasion_story
				value = yes
			}
		}
		invader_story_on_death_effect = { INVADER_DYNN = rurikid }
	}

	# maintenance
	effect_group = {
		days = historicinvasions_story_maintenance_value

		trigger = { always = yes }

		# make sure they have money on hand
		triggered_effect = {
			trigger = { story_owner = { gold < 0 } }
			effect = { story_owner = { add_gold = 150 } }
		}

		# end the story prematurely, because their conquests have been completed
		triggered_effect = {
			trigger = {
				story_owner = { completely_controls_region = special_rurikid_conquest_region_total }
			}
			effect = {
				debug_log = "ending rurikid story because the conquest goal has been completed"
				end_story = yes
			}
		}

		# end the story prematurely, because the invader has lost their #th offensive war
		triggered_effect = {
			trigger = {
				NOT = { has_game_rule = historicinvasions_invasion_ends_after_defeat_disabled }
			}
			effect = {
				historicinvasions_invasion_ends_after_defeat_effect = yes
			}
		}
	}

	#Start a new war if none is ongoing
	effect_group = {
		days = { 365 1095 }
		#chance = 100

		trigger = { story_owner = { is_ai = yes } }

		first_valid = {
			triggered_effect = {
				trigger = {
					story_owner = { has_character_flag = invasion_ends_after_player_win }
				}
				effect = {
					debug_log = "ending rurikid story because a player was able to defend against the invader"
					end_story = yes
				}
			}
			triggered_effect = {
				trigger = { story_owner = { should_invader_declare_war_trigger = { INVADER_TITLE = scope:story.var:invader_title_var } } }
				effect = {
					story_owner = {
						war_target_evaluation_and_declaration_effect = {
							INVADER_DYNN = rurikid
							INVADER_CHAR = scope:story.story_owner
							INVADER_REGION = total
							INVADER_TARGET_TITLE_TIER = kingdom
							INVADER_RANGE = neighboring
							INVADER_LIEGE_RESTRICTIONS = "NOT = { this = title:k_magyar.holder }"
						}
					}
				}
			}
		}
	}

	# Story handling in case of succession weirdness
	effect_group = {
		days = historicinvasions_succession_weirdness_value

		trigger = { NOT = { story_owner = { has_title = scope:story.var:invader_title_var } } }

		# either transfer story ownership or end story if the title is not created
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				hi_check_for_valid_story_owner_effect = {
					INVADER_TITLE = scope:story.var:invader_title_var
					STORY_COUNTER_INCREASE = 0
				}
			}
		}
	}

	# switch to local title
	effect_group = {
		days = 240

		trigger = {}

		triggered_effect = {
			trigger = {
				hi_can_story_switch_to_local_title_trigger = {
					INVADER_TITLE = e_rurikid
					LOCAL_TITLE = e_russia
				}
			}
			effect = {
				hi_story_switches_to_local_title_effect = {
					INVADER_TITLE = e_rurikid
					LOCAL_TITLE = e_russia
				}
				title:e_russia = {
					set_title_name = e_rurikid
					set_coa = e_rurikid
				}
			}
		}
	}
}

story_rurikid_conversion = {
	on_setup = {
		if = {
			limit = {
				story_owner.faith = { NOT = { has_doctrine_parameter = unreformed } }
			}
			debug_log = "ending rurikid conversion story prematurely because its starting conditions were not met"
			end_story = yes
		}
		# start conversion events
		else = {
			story_owner = {
				trigger_event = {
					id = rurikid_invasion.5001
					days = 1
				}
			}
		}
	}
	on_end = {}
	on_owner_death = {
		if = {
			limit = { exists = story_owner.player_heir }
			make_story_owner = story_owner.player_heir
		}
		else = {
			debug_log = "rurikid conversion story ended because a valid heir could not be found"
			end_story = yes
		}
	}

	effect_group = {
		years = 1
		trigger = {
			story_owner = {
				faith = { NOT = { has_doctrine_parameter = unreformed } }
			}
		}
		triggered_effect = {
			trigger = {}
			effect = {
				end_story = yes
			}
		}
	}
}
