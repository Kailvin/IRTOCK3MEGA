# Story for Chola dynasty
# originally made by Petter Vilberg
# adjusted by me

story_chola_invasion = {
	on_setup = {
		historicinvasions_story_setup_effect = { # global start variable, story variables, conditional dissolution protection
			INVADER_DYNN = chola # for story start global variable
			INVADER_TITLE = title:e_chola # title the invader spawns with
		}
		story_owner = {
			chola_maintenance_effect = {
				INVADER_DYNN = chola
				INVADER_CHAR = scope:story.story_owner
				INVADER_TITLE = scope:story.var:invader_title_var
				LANDLESS_YES_NO = no
				STARTING_TIER = duchy
			}
			trigger_event = {
				id = chola_invasion.0200
				months = 1
			}
		}
		scope:story.var:invader_title_var.title_capital_county = {
			add_county_modifier = {
				modifier = historicinvasions_county_tier_two_modifier
				years = 100
			}
		}
	}

	on_end = {
		invader_story_on_end_effect = { INVADER_DYNN = chola }

		# establishing kingdoms will fail if the story ended because the story owner died (can't grant titles to dead characters); that's why the script tries to create them periodically further down
		story_owner = { create_chola_successor_kingdoms_effect = yes }
		scope:story.var:invader_title_var = { set_destroy_on_succession = yes }
	}

	on_owner_death = {
		# random chance to set variable to end story after ruler N dies, depending on how many story owners there have been in total
		random = {
			chance = 0
			modifier = {
				add = 0
				var:story_owner_counter >= 1 # end story after death of ruler N?
			}
			modifier = {
				add = 10
				var:story_owner_counter >= 2
			}
			modifier = {
				add = 20
				var:story_owner_counter >= 3
			}
			modifier = {
				add = 50
				var:story_owner_counter >= 4
			}
			modifier = {
				add = 20
				var:story_owner_counter >= 5
			}
			set_variable = {
				name = end_of_invasion_story
				value = yes
			}
		}
		invader_story_on_death_effect = { INVADER_DYNN = chola }
	}

	# maintenance
	effect_group = {
		days = historicinvasions_story_maintenance_value

		trigger = { always = yes }

		# make sure they have money on hand
		triggered_effect = {
			trigger = { story_owner = { gold < 0 } }
			effect = { story_owner = { add_gold = 200 } }
		}

		# end the story prematurely, because their conquests have been completed
		triggered_effect = {
			trigger = {
				story_owner = { completely_controls_region = special_chola_conquest_region_total }
			}
			effect = {
				debug_log = "ending chola story because the conquest goal has been completed"
				end_story = yes
			}
		}

		# end the story prematurely, because the invader has lost their #th offensive war
		triggered_effect = {
			trigger = {
				NOT = { has_game_rule = historicinvasions_invasion_ends_after_defeat_disabled }
			}
			effect = {
				historicinvasions_invasion_ends_after_defeat_effect = yes
			}
		}
	}

	#Start a new war if none is ongoing
	effect_group = {
		days = { 180 365 }

		trigger = { story_owner = { is_ai = yes } }

		first_valid = {
			triggered_effect = {
				trigger = {
					story_owner = { has_character_flag = invasion_ends_after_player_win }
				}
				effect = {
					debug_log = "ending chola story because a player was able to defend against the invader"
					end_story = yes
				}
			}
			# solution: directly jump into the duchy
			triggered_effect = {
				trigger = {
					# is asia addon installed?
					is_historicinvasions-rajasofasia_active = yes
					story_owner = {
						NOT = { completely_controls_region = special_chola_conquest_region_4 } # doesn't already own land in the targeted region
						should_invader_declare_war_trigger = { INVADER_TITLE = scope:story.var:invader_title_var }
						completely_controls_region = special_chola_conquest_region_start # has consolidated southern India
					}
				}
				effect = {
					story_owner = { chola_conquer_kedah_addon_effect = yes }
				}
			}
			# problem: the distance between southern India and Kedah is too large - they are not neighboring!
			triggered_effect = {
				trigger = { story_owner = { should_invader_declare_war_trigger = { INVADER_TITLE = scope:story.var:invader_title_var } } }
				effect = {
					story_owner = {
						war_target_evaluation_and_declaration_effect = {
							INVADER_DYNN = chola
							INVADER_CHAR = scope:story.story_owner
							INVADER_REGION = total
							INVADER_TARGET_TITLE_TIER = kingdom
							INVADER_RANGE = neighboring
							INVADER_LIEGE_RESTRICTIONS = ""
						}
					}
				}
			}
		}
	}

	# Story handling in case of succession weirdness
	effect_group = {
		days = historicinvasions_succession_weirdness_value

		trigger = { NOT = { story_owner = { has_title = scope:story.var:invader_title_var } } }

		# either transfer story ownership or end story if the title is not created
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				hi_check_for_valid_story_owner_effect = {
					INVADER_TITLE = scope:story.var:invader_title_var
					STORY_COUNTER_INCREASE = 0
				}
			}
		}
	}

	# hand out successor kingdoms prematurely
	effect_group = {
		days = 140

		trigger = { always = yes }

		triggered_effect = {
			trigger = {
				OR = {
					title:k_tamilakam = { is_title_created = no }
					title:k_andhra = { is_title_created = no }
					title:k_lanka = { is_title_created = no }
				}
			}
			effect = { story_owner = { create_chola_successor_kingdoms_effect = yes } }
		}
	}
}
