### GOLDEN HORDE
story_hulagu_invasion = {
	on_setup = {
		historicinvasions_story_setup_effect = { # global start variable, story variables, conditional dissolution protection
			INVADER_DYNN = hulagu # for story start global variable
			INVADER_TITLE = title:e_ilkhanate # title the invader spawns with
		}
		story_owner = {
			hulagu_maintenance_effect = {
				INVADER_DYNN = hulagu
				INVADER_CHAR = scope:story.story_owner
				INVADER_TITLE = scope:story.var:invader_title_var
				LANDLESS_YES_NO = no
			}
			trigger_event = {
				id = hulagu_invasion.0200
				months = 1
			}
		}
		scope:story.var:invader_title_var.title_capital_county = {
			add_county_modifier = {
				modifier = historicinvasions_county_tier_two_modifier
				years = 75
			}
		}
	}

	on_end = {
		invader_story_on_end_effect = { INVADER_DYNN = hulagu }
		### clean up potential de jure border gore if invader wasn't very successful; check de jure shift triggered effect!
		# persia
		if = {
			limit = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count <= 2 } }
				title:e_persia = { is_titular = no }
			}
			scope:story.var:invader_title_var = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					set_de_jure_liege_title = title:e_persia
				}
			}
		}
		# seljuk
		else_if = {
			limit = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count <= 2 } }
				title:e_seljuk = { is_titular = no }
			}
			scope:story.var:invader_title_var = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					set_de_jure_liege_title = title:e_seljuk
				}
			}
		}
		# anushtiginid
		else_if = {
			limit = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count <= 2 } }
				title:e_anushtiginid = { is_titular = no }
			}
			scope:story.var:invader_title_var = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					set_de_jure_liege_title = title:e_anushtiginid
				}
			}
		}
		# timurid
		else_if = {
			limit = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count <= 2 } }
				title:e_timurid = { is_titular = no }
			}
			scope:story.var:invader_title_var = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					set_de_jure_liege_title = title:e_timurid
				}
			}
		}
		# fallback
		else = {
			debug_log = "on story end: invader de jure land not reverted"
		}
	}

	on_owner_death = {
		# random chance to set variable to end story after ruler N dies, depending on how many story owners there have been in total
		random = {
			chance = 0
			modifier = {
				add = 10
				var:story_owner_counter >= 1 # end story after death of ruler N?
			}
			modifier = {
				add = 80
				var:story_owner_counter >= 2
			}
			modifier = {
				add = 10
				var:story_owner_counter >= 3
			}
			set_variable = {
				name = end_of_invasion_story
				value = yes
			}
		}
		invader_story_on_death_effect = { INVADER_DYNN = hulagu }
	}

	# maintenance
	effect_group = {
		days = historicinvasions_story_maintenance_value

		trigger = { always = yes }

		# make sure they have money on hand
		triggered_effect = {
			trigger = { story_owner = { gold < 0 } }
			effect = { story_owner = { add_gold = 500 } }
		}

		# end the story prematurely, because their conquests have been completed
		triggered_effect = {
			trigger = {
				story_owner = { completely_controls_region = special_hulagu_conquest_region_total }
			}
			effect = {
				debug_log = "ending hulagu story because the conquest goal has been achieved"
				end_story = yes
			}
		}

		# end the story prematurely, because the invader has lost their #th offensive war
		triggered_effect = {
			trigger = {
				NOT = { has_game_rule = historicinvasions_invasion_ends_after_defeat_disabled }
			}
			effect = {
				historicinvasions_invasion_ends_after_defeat_effect = yes
			}
		}
	}

	#Start a new war if none is ongoing
	effect_group = {
		days = { 200 400 }

		trigger = {
			story_owner = {
				is_ai = yes
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = {
					story_owner = { has_character_flag = invasion_ends_after_player_win }
				}
				effect = {
					debug_log = "ending hulagu story because a player was able to defend against the invader"
					end_story = yes
				}
			}

			### war declaration
			triggered_effect = {
				trigger = { story_owner = { should_invader_declare_war_trigger = { INVADER_TITLE = scope:story.var:invader_title_var } } }
				effect = {
					story_owner = {
						war_target_evaluation_and_declaration_effect = {
							INVADER_DYNN = hulagu
							INVADER_CHAR = scope:story.story_owner
							INVADER_REGION = total
							INVADER_TARGET_TITLE_TIER = kingdom
							INVADER_RANGE = neighboring
							INVADER_LIEGE_RESTRICTIONS = ""
						}
					}
				}
			}
		}
	}

	# Story handling in case of succession weirdness
	effect_group = {
		days = historicinvasions_succession_weirdness_value

		trigger = { NOT = { story_owner = { has_title = scope:story.var:invader_title_var } } }

		# either transfer story ownership or end story if the title is not created
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				hi_check_for_valid_story_owner_effect = {
					INVADER_TITLE = scope:story.var:invader_title_var
					STORY_COUNTER_INCREASE = 0
				}
			}
		}
	}

	# de jure updates
	effect_group = {
		days = 240
		trigger = { # account for struggle for Persia
			NOT = {
				title:d_isfahan.empire = title:e_arabia
			}
			title:e_persia = { is_titular = no }
		}
		### gradual
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				historicinvasions_de_jure_change = {
					INVADER_DYNN = hulagu
					INVADER_CHAR = scope:story.story_owner
					INVADER_TITLE = scope:story.var:invader_title_var
					INVADER_TARGET_TITLE_TIER = kingdom
					DE_JURE_PERCENT = 0.7
				}
			}
		}
		### instant merge if big enough
		# persia
		triggered_effect = {
			trigger = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count > 2 } }
				title:e_persia = { is_titular = no }
			}
			effect = {
				title:e_persia = {
					every_in_de_jure_hierarchy = {
						limit = { tier = tier_kingdom }
						set_de_jure_liege_title = scope:story.var:invader_title_var
					}
				}
			}
		}
		# seljuk
		triggered_effect = {
			trigger = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count > 2 } }
				title:e_seljuk = { is_titular = no }
			}
			effect = {
				title:e_seljuk = {
					every_in_de_jure_hierarchy = {
						limit = { tier = tier_kingdom }
						set_de_jure_liege_title = scope:story.var:invader_title_var
					}
				}
			}
		}
		# anushtiginid
		triggered_effect = {
			trigger = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count > 2 } }
				title:e_anushtiginid = { is_titular = no }
			}
			effect = {
				title:e_anushtiginid = {
					every_in_de_jure_hierarchy = {
						limit = { tier = tier_kingdom }
						set_de_jure_liege_title = scope:story.var:invader_title_var
					}
				}
			}
		}
		# timurid
		triggered_effect = {
			trigger = {
				scope:story.var:invader_title_var = { any_in_de_jure_hierarchy = { tier = tier_kingdom count > 2 } }
				title:e_timurid = { is_titular = no }
			}
			effect = {
				title:e_timurid = {
					every_in_de_jure_hierarchy = {
						limit = { tier = tier_kingdom }
						set_de_jure_liege_title = scope:story.var:invader_title_var
					}
				}
			}
		}
	}
}
