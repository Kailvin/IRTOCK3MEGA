namespace = ziyarid_invasion

### Spawn invader with the associated story...
ziyarid_invasion.0001 = {
	scope = none
	hidden = yes
	
	trigger = {
		exists = title:k_ziyarid # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_ziyarid_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "ziyarid story has failed to trigger, because a player directly controls all counties within the starting region; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "ziyarid story owner has appeared"
		debug_log_date = yes
		
		spawn_ziyarid_invader_effect = {
			INVADER_DYNN = ziyarid
		}
		scope:invader = {
			create_story = story_ziyarid_invasion
		}
	}
}

# Event for handling new army spawning on succession
ziyarid_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_ziyarid_troops_effect = yes
	}
}

# Event for declaring the very first war; values like current_ / max_military_strength aren't immediately filled with a value and return 0 instead
ziyarid_invasion.0200 = {
	hidden = yes
	trigger = { is_ai = yes } # don't force a player into a war
	immediate = {
		war_target_evaluation_and_declaration_effect = {
			INVADER_DYNN = ziyarid
			INVADER_CHAR = scope:story.story_owner
			INVADER_REGION = total
			INVADER_TARGET_TITLE_TIER = duchy
			INVADER_RANGE = neighboring
			INVADER_LIEGE_RESTRICTIONS = ""
		}
	}
}

### The invader has appeared! Would you like to play as them?
ziyarid_invasion.1001 = { # only players see this
	type = character_event
	title = ziyarid_invasion.1001.t
	#desc = ziyarid_invasion.1001.desc

	desc = {
		desc = ziyarid_invasion.1001.desc.intro
		triggered_desc = { # Zoroastrian uprising
			trigger = {
				any_ruler = {
					count = 0
					is_landed = yes
					capital_province = { geographical_region = special_historicinvasions_persia_region }
					is_independent_ruler = yes
					religion = religion:zoroastrianism_religion
					NOT = { this = scope:invader }
				}
			}
			desc = ziyarid_invasion.1001.desc.zoroastrian_uprising
		}
		triggered_desc = { # there are already Zoroastrian rulers present
			trigger = {
				any_ruler = {
					count > 0
					is_landed = yes
					capital_province = { geographical_region = special_historicinvasions_persia_region }
					is_independent_ruler = yes
					religion = religion:zoroastrianism_religion
					NOT = { this = scope:invader }
				}
			}
			desc = ziyarid_invasion.1001.desc.other_zoroastrians
		}
		desc = ziyarid_invasion.1001.desc.outro
	}
	
	theme = hi_theme_mountains
	
	left_portrait = {
		character = scope:invader
		animation = personality_zealous
	}

	trigger = { hi_show_invader_notification_event_trigger = yes }
	
	immediate = {
		character:159601 = { save_scope_as = sassanid_dynasty_member }

		# localization
		if = {
			limit = {
				any_ruler = {
					count > 0
					is_landed = yes
					capital_province = { geographical_region = special_historicinvasions_persia_region }
					is_independent_ruler = yes
					religion = religion:zoroastrianism_religion
					NOT = { this = scope:invader }
				}
			}
			random_ruler = {
				limit = {
					is_landed = yes
					capital_province = { geographical_region = special_historicinvasions_persia_region }
					is_independent_ruler = yes
					religion = religion:zoroastrianism_religion
					NOT = { this = scope:invader }
				}
				save_scope_as = other_zoroastrian_ruler
			}
		}
	}

	option = { name = ziyarid_invasion.1001.a }

	option = { # play as the newly spawned invader
		name = ziyarid_invasion.1001.b
		trigger = { hi_play_as_invader_event_option_trigger = yes }
		add_internal_flag = dangerous
		scope:invader = {
			set_variable = {
				name = historicinvasions_previous_ruler_var
				value = root
				months = 1
			}
		}
		set_player_character = scope:invader
	}
}
