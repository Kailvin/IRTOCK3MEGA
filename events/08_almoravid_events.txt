namespace = almoravid_invasion

### Spawn invader with the associated story...
almoravid_invasion.0001 = {
	scope = none
	hidden = yes
	
	trigger = {
		exists = title:e_almoravid # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_almoravid_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "almoravid story has failed to trigger; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "almoravid story owner has appeared"
		debug_log_date = yes
		
		character:273081 = { save_scope_as = talakakin_lamtuna } # founder of dynasty lamtuna
		character:155077 = { save_scope_as = turjut_almoravid } # founder of house almoravid

		if = {
			limit = { game_start_date >= 973.1.1 } # birth of Turjut
			scope:turjut_almoravid.house = {
				# pick an already existing house member
				ordered_house_member = {
					limit = {
						is_landed = yes
						highest_held_title_tier > tier_barony
						capital_county = { title_province = { geographical_region = special_almoravid_conquest_region_total } }
						is_independent_ruler = yes
						is_adult = yes
						is_healthy = yes
						is_imprisoned = no
					}
					alternative_limit = {
						is_landed = yes
						highest_held_title_tier > tier_barony
						capital_county = { title_province = { geographical_region = special_almoravid_conquest_region_total } }
						is_adult = yes
						is_healthy = yes
						is_imprisoned = no
					}
					order_by = realm_size
					save_scope_as = invader
				}
			}
		}

		spawn_almoravid_invader_effect = {
			INVADER_DYNN = almoravid
		}
		scope:invader = {
			create_story = story_almoravid_invasion
		}
	}
}

# Event for handling army spawning
almoravid_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_almoravid_troops_effect = yes
	}
}

# Event for declaring the very first war; values like current_ / max_military_strength aren't immediately filled with a value and return 0 instead
almoravid_invasion.0200 = {
	hidden = yes
	trigger = { is_ai = yes } # don't force a player into a war
	immediate = {
		war_target_evaluation_and_declaration_effect = {
			INVADER_DYNN = almoravid
			INVADER_CHAR = scope:story.story_owner
			INVADER_REGION = total
			INVADER_TARGET_TITLE_TIER = kingdom
			INVADER_RANGE = neighboring
			INVADER_LIEGE_RESTRICTIONS = ""
		}
	}
}

### The invader has appeared! Would you like to play as them?
almoravid_invasion.1001 = { # only players see this
	type = character_event
	title = almoravid_invasion.1001.t
	desc = almoravid_invasion.1001.desc
	
	theme = hi_theme_mountains

	left_portrait = {
		character = scope:invader
		animation = personality_zealous
	}

	trigger = { hi_show_invader_notification_event_trigger = yes }
	
	option = { name = almoravid_invasion.1001.a }

	option = { # play as the newly spawned invader
		name = almoravid_invasion.1001.b
		trigger = { hi_play_as_invader_event_option_trigger = yes }
		add_internal_flag = dangerous
		scope:invader = {
			set_variable = {
				name = historicinvasions_previous_ruler_var
				value = root
				months = 1
			}
		}
		set_player_character = scope:invader
	}
}
