namespace = sabbah_invasion

### Spawn invader with the associated story...
sabbah_invasion.0001 = {
	scope = none
	hidden = yes

	trigger = {
		exists = title:k_sabbah # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_sabbah_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "sabbah story has failed to trigger; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "sabbah story owner has appeared"
		debug_log_date = yes
		
		spawn_sabbah_invader_effect = {
			INVADER_DYNN = sabbah
		}
		scope:invader = {
			create_story = story_sabbah_invasion
		}
	}
}

# Event for handling army spawning
sabbah_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_sabbah_troops_effect = yes
	}
}

### The invader has appeared! Would you like to play as them?
sabbah_invasion.1001 = { # only players see this
	type = character_event
	title = sabbah_invasion.1001.t
	desc = sabbah_invasion.1001.desc
	
	theme = hi_theme_mountains

	left_portrait = {
		character = scope:invader
		animation = personality_bold
	}

	trigger = { hi_show_invader_notification_event_trigger = yes }
	
	option = { name = sabbah_invasion.1001.a }

	option = { # play as the newly spawned invader
		name = sabbah_invasion.1001.b
		trigger = { hi_play_as_invader_event_option_trigger = yes }
		custom_tooltip = invader_title_destroyed_on_different_faith_tt
		add_internal_flag = dangerous
		scope:invader = {
			set_variable = {
				name = historicinvasions_previous_ruler_var
				value = root
				months = 1
			}
		}
		set_player_character = scope:invader
	}
}
