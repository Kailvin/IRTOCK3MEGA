namespace = qarghuyid_invasion

### Spawn invader with the associated story...
qarghuyid_invasion.0001 = {
	scope = none
	hidden = yes
	
	trigger = {
		exists = title:d_qarghuyid # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_qarghuyid_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "qarghuyid story has failed to trigger; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "qarghuyid story owner has appeared"
		debug_log_date = yes
				
		spawn_qarghuyid_invader_effect = {
			INVADER_DYNN = qarghuyid
		}
		scope:invader = {
			create_story = story_qarghuyid_invasion
		}
	}
}

# Event for handling new army spawning on succession
qarghuyid_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_qarghuyid_troops_effect = yes
	}
}

# Event for declaring the very first war; values like current_ / max_military_strength aren't immediately filled with a value and return 0 instead
qarghuyid_invasion.0200 = {
	hidden = yes
	orphan = yes
	trigger = { is_ai = yes } # don't force a player into a war
	immediate = {}
}

### The invader has appeared! Would you like to play as them?
qarghuyid_invasion.1001 = { # only players see this
	type = character_event
	title = qarghuyid_invasion.1001.t
	desc = qarghuyid_invasion.1001.desc
	
	theme = hi_theme_desert

	left_portrait = {
		character = scope:invader
		animation = personality_bold
	}

	trigger = { hi_show_invader_notification_event_trigger = yes }
	
	option = { name = qarghuyid_invasion.1001.a }

	option = { # play as the newly spawned invader
		name = qarghuyid_invasion.1001.b
		trigger = { hi_play_as_invader_event_option_trigger = yes }
		custom_tooltip = invader_title_destroyed_on_succession_tt
		add_internal_flag = dangerous
		scope:invader = {
			set_variable = {
				name = historicinvasions_previous_ruler_var
				value = root
				months = 1
			}
		}
		set_player_character = scope:invader
	}
}
