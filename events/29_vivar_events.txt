namespace = vivar_invasion

### Spawn invader with the associated story...
vivar_invasion.0001 = {
	scope = none
	hidden = yes
	
	trigger = {
		exists = title:d_vivar # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_vivar_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "vivar story has failed to trigger; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "vivar story owner has appeared"
		debug_log_date = yes
				
		spawn_vivar_invader_effect = {
			INVADER_DYNN = vivar
		}
		scope:invader = {
			create_story = story_vivar_invasion
		}
	}
}

# Event for handling new army spawning on succession
vivar_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_vivar_troops_effect = yes
	}
}

# Event for declaring the very first war; values like current_ / max_military_strength aren't immediately filled with a value and return 0 instead
vivar_invasion.0200 = {
	hidden = yes
	orphan = yes
	trigger = { is_ai = yes } # don't force a player into a war
	immediate = {}
}
