namespace = bearn_invasion

### Spawn invader with the associated story...
bearn_invasion.0001 = {
	scope = none
	hidden = yes
	
	trigger = {
		exists = title:d_bearn # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_bearn_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "bearn story has failed to trigger; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "bearn story owner has appeared"
		debug_log_date = yes
				
		spawn_bearn_invader_effect = {
			INVADER_DYNN = bearn
		}
		scope:invader = {
			create_story = story_bearn_invasion
		}
	}
}

# Event for handling new army spawning on succession
bearn_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_bearn_troops_effect = yes
	}
}

# Event for declaring the very first war; values like current_ / max_military_strength aren't immediately filled with a value and return 0 instead
bearn_invasion.0200 = {
	hidden = yes
	orphan = yes
	trigger = { is_ai = yes } # don't force a player into a war
	immediate = {}
}

### The invader has appeared! Would you like to play as them?
bearn_invasion.1001 = { # only players see this
	type = character_event
	title = bearn_invasion.1001.t
	desc = bearn_invasion.1001.desc
	
	theme = hi_theme_farmlands

	left_portrait = {
		character = scope:invader
		animation = personality_bold
	}

	trigger = { hi_show_invader_notification_event_trigger = yes }
	
	immediate = {
		title:k_england.holder ?= { save_scope_as = ruler_england }
		title:e_byzantium.holder ?= { save_scope_as = ruler_byzantium }
		title:b_constantinople.holder ?= { save_scope_as = ruler_constantinople }
	}

	option = { name = bearn_invasion.1001.a }

	option = { # play as the newly spawned invader
		name = bearn_invasion.1001.b
		trigger = { hi_play_as_invader_event_option_trigger = yes }
		custom_tooltip = invader_title_destroyed_on_succession_tt
		add_internal_flag = dangerous
		scope:invader = {
			set_variable = {
				name = historicinvasions_previous_ruler_var
				value = root
				months = 1
			}
		}
		set_player_character = scope:invader
	}
}
