namespace = song_invasion

### Spawn invader with the associated story...
song_invasion.0001 = {
	scope = none
	hidden = yes

	trigger = {
		exists = title:e_song # update compatibility - new titles in old saves return false
		any_county_in_region = {
			region = special_song_conquest_region_start
			count > 0
			holder = { is_ai = yes }
		}
	}

	on_trigger_fail = {
		debug_log = "song story has failed to trigger; trying again next year"
		debug_log_date = yes
	}

	immediate = {
		debug_log = "song story owner has appeared"
		debug_log_date = yes
		
		spawn_song_invader_effect = {
			INVADER_DYNN = song
		}
		scope:invader = {
			create_story = story_song_invasion
		}
	}
}

# Event for handling army spawning
song_invasion.0100 = {
	hidden = yes
	immediate = {
		spawn_song_troops_effect = yes
	}
}

# Event for declaring the very first war; values like current_ / max_military_strength aren't immediately filled with a value and return 0 instead
song_invasion.0200 = {
	hidden = yes
	trigger = { is_ai = yes } # don't force a player into a war
	immediate = {
		change_government = administrative_government
		set_variable = {
            name = mil_comp
            value = 1500
        }
        set_variable = {
            name = civic_comp
            value = 1500
        }
        set_variable = {
            name = diplo_comp
            value = 1500
        }
		war_target_evaluation_and_declaration_effect = {
			INVADER_DYNN = song
			INVADER_CHAR = scope:story.story_owner
			INVADER_REGION = total
			INVADER_TARGET_TITLE_TIER = kingdom
			INVADER_RANGE = neighboring
			INVADER_LIEGE_RESTRICTIONS = ""
		}
	}
}

### The invader has appeared! Would you like to play as them?
song_invasion.1001 = { # only players see this
	type = character_event
	title = song_invasion.1001.t
	desc = song_invasion.1001.desc
	
	theme = hi_theme_jungle

	left_portrait = {
		character = scope:invader
		animation = personality_bold
	}

	trigger = { hi_show_invader_notification_event_trigger = yes }
	
	option = { name = song_invasion.1001.a }

	option = { # play as the newly spawned invader
		name = song_invasion.1001.b
		trigger = { hi_play_as_invader_event_option_trigger = yes }
		add_internal_flag = dangerous
		scope:invader = {
			set_variable = {
				name = historicinvasions_previous_ruler_var
				value = root
				months = 1
			}
		}
		set_player_character = scope:invader
	}
}
