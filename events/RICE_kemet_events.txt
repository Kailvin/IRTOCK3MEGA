# Come across a village celebrating a festival to Osiris
VIETkemet.0001 = {
	type = character_event
	title = VIETkemet.0001.t
	desc = VIETkemet.0001.desc
	theme = party
	override_background = { reference = RICE_kemet_background_egypt_village }
	
	left_portrait = root
	
	trigger = {
		capital_province = {
			geographical_region = ghw_region_egypt_et_al
		}	
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = trusting
		}
		modifier = {
			add = -0.2
			has_trait = paranoid
		}
		modifier = {
			add = 0.2
			has_trait = lazy
		}
		modifier = {
			add = -0.2
			has_trait = diligent
		}
	}
	
	option = { # Observe as a ruler
		name = VIETkemet.0001.a	
		add_prestige = 25
		trigger_event = VIETkemet.0002
	}	
	option = { # Disguise as a commoner
		name = VIETkemet.0001.b	
		add_prestige = -50
		add_piety = 50
		stress_impact = {
			shy = minor_stress_impact_gain
			paranoid = minor_stress_impact_gain
			arrogant = minor_stress_impact_gain
		}
		trigger_event = VIETkemet.0003
	}	
	option = { # they should serve me!
		name = VIETkemet.0001.c
		add_dread = 10
		add_prestige = 50
		add_piety = -50
		stress_impact = {
			compassionate = minor_stress_impact_gain
			zealous = minor_stress_impact_gain
			humble = minor_stress_impact_gain
			generous = minor_stress_impact_gain
			just = minor_stress_impact_gain
		}
	}
}

# Given an amulet of Khnum
VIETkemet.0007 = {
	type = character_event
	title = VIETkemet.0007.t
	desc = VIETkemet.0007.desc
	theme = realm
	override_icon = { reference = "gfx/interface/icons/event_types/type_faith.dds" }
	
	left_portrait = {
		character = scope:khnum_gifter
		animation = personality_forgiving
	}
	
	trigger = {
		any_courtier = {
			is_available_ai_adult = yes
			faith = root.faith		
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = gregarious
		}
		modifier = {
			add = -0.2
			has_trait = shy
		}
		modifier = {
			add = 0.2
			has_trait = patient
		}
		modifier = {
			add = -0.2
			has_trait = impatient
		}
		modifier = {
			add = 0.2
			has_trait = just
		}
		modifier = {
			add = -0.2
			has_trait = arbitrary
		}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				is_available_ai_adult = yes
				faith = root.faith
			}
			save_scope_as = khnum_gifter
		}
	}
	
	option = { # ok
		name = VIETkemet.0007.a
		stress_impact = {
			paranoid = minor_stress_impact_gain
			callous = minor_stress_impact_gain
			cynical = miniscule_stress_impact_gain
			arrogant = miniscule_stress_impact_gain
		}
		add_character_modifier = {
			modifier = RICE_kemet_khnum_amulet
			years = 10
		}
		scope:khnum_gifter = {
			add_opinion = {
				target = root
				modifier = RICE_kemet_opinion_accepted_gift
			}		
		}
	}	
	option = { # ok
		name = VIETkemet.0007.b
		stress_impact = {
			greedy = minor_stress_impact_gain
			zealous = minor_stress_impact_gain
			trusting = minor_stress_impact_gain
		}
		scope:khnum_gifter = {
			add_opinion = {
				target = root
				modifier = RICE_kemet_opinion_didnt_accept_gift
			}		
		}
	}
}
